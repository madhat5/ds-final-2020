                .domain([0, 1, 2])

                    let minutes = (+a[0]) * 60 + (+a[1]);

                    if (minutes < 420) {

                    } else if (minutes >= 420 && minutes <= 480) {

                    }

                    return y(d.temp);

                    return z(d.weight);

                    let hm = d.hoursSlept; // your input string

                        return myColor(0);
